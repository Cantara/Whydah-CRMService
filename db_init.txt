### User, schema and table definitions

DROP ROLE crmadmin;
CREATE ROLE crmadmin LOGIN PASSWORD 'secret'  NOINHERIT, CREATEDB;

DROP DATABASE crmservice;
\set role crmadmin
CREATE DATABASE crmservice OWNER crmadmin;
\connect crmservice

CREATE TABLE customers (
   customer_id    CHAR(256) PRIMARY KEY      NOT NULL,
   data           CHAR(5000) NOT NULL
);

CREATE TABLE profileimages (
   customer_id    CHAR(256) PRIMARY KEY      NOT NULL,
   image BYTEA,
   contenttype    CHAR(256)
);

GRANT ALL ON customers TO crmadmin;
GRANT ALL ON profileimages TO crmadmin;
